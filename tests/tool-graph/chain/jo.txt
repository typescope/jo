dep a:
  dir = a
  sast = a/.build/a/sast
  kind = check
dep b:
  dir = b
  sast = b/.build/b/sast
  kind = check
check-libs = [a/.build/a/sast, b/.build/b/sast]
link-libs  = []

# lib: a
jo compile --sast a/src/Main.jo -d a/.build/a/sast

# lib: b
jo compile --sast b/src/Main.jo --lib a/.build/a/sast -d b/.build/b/sast

# root (app)
jo compile --python src/Main.jo --lib a/.build/a/sast --lib b/.build/b/sast -o .build/jo/target/app.py
