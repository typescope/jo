dep agent-api:
  dir = api
  sast = api/.build/agent-api/sast
  kind = check
dep agent-runtime:
  dir = runtime
  sast = runtime/.build/agent-runtime/sast
  kind = link
check-libs = [api/.build/agent-api/sast]
link-libs  = [runtime/.build/agent-runtime/sast]

# lib: agent-api
jo compile --sast api/src/Main.jo -d api/.build/agent-api/sast

# lib: agent-runtime
jo compile --sast runtime/src/Main.jo --lib api/.build/agent-api/sast -d runtime/.build/agent-runtime/sast

# root (app)
jo compile --python src/Main.jo --lib api/.build/agent-api/sast --link-lib runtime/.build/agent-runtime/sast -o .build/jo/target/my-agent.py
